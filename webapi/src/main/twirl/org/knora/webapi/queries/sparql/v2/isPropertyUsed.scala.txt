@import org.knora.webapi.IRI

@*
 * Checks whether a property is used (i.e., is the property of any statements).
 *
 * @param propertyIri the IRI of the property.
 * @param classIri the IRI of class to check.
 *@
@(propertyIri: org.knora.webapi.messages.SmartIri, classIri: org.knora.webapi.messages.SmartIri)

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX knora-base: <http://www.knora.org/ontology/knora-base#>

ASK

WHERE {
    BIND(IRI("@{propertyIri.toString}") AS ?property)
    BIND(IRI("@{classIri.toString}") AS ?classIri)

    {
        # select all items of type classIri with property ?property.
        ?s ?property ?o .
        ?s rdf:type ?classIri .
    }

    UNION {
            # select all items belonging to a subclass of classIri and with property ?property.
            ?s ?property ?o .
            ?s rdf:type ?class .
            ?class rdfs:subClassOf* ?classIri .
          }
}
