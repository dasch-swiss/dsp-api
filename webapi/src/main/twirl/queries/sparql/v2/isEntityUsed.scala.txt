@import org.knora.webapi.util.SmartIri

@*
 * Checks whether an ontology entity is used (i.e. is the object
 * of any statements).
 *
 * @param triplestore the name of the triplestore being used.
 * @param entityIri the IRI of the ontology.
 * @param checkDataOnly if true, only data (not ontology entities) will be checked.
 *@
@(triplestore: String,
  entityIri: SmartIri,
  checkDataOnly: Boolean)

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX knora-base: <http://www.knora.org/ontology/knora-base#>

SELECT DISTINCT ?isUsed
@* Ensure that inference is not used in this query. *@
@if(triplestore.startsWith("graphdb")) {
    FROM <http://www.ontotext.com/explicit>
}
WHERE {
    BIND(IRI("@entityIri") AS ?entity)
    BIND(true AS ?isUsed)

    ?s ?p ?entity .

    @if(checkDataOnly) {

        ?s rdf:type ?sType .

        FILTER(!(?sType = owl:Class || ?sType = owl:Restriction || ?sType = owl:ObjectProperty || ?sType = owl:DatatypeProperty || ?sType = owl:AnnotationProperty || ?sType = rdf:Property))

    }
}
