@import org.knora.webapi.IRI

@**
 * Gets all information about a single project, given the project's knora-base:projectShortname.
 *
 * @param triplestore the name of the triplestore being used.
 * @param shortname the project's `knora-base:projectShortname`.
 *@
@(triplestore: String,
  shortname: String)

prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#>
prefix knora-base: <http://www.knora.org/ontology/knora-admin#>

SELECT ?s ?p ?o
@* Ensure that inference is not used in this query. *@
@if(triplestore.startsWith("graphdb")) {
    FROM <http://www.ontotext.com/explicit>
}
WHERE {
    BIND("@shortname" as ?shortname)

    ?s rdf:type knora-base:knoraProject .
    ?s knora-base:projectShortname ?shortname .
    ?s ?p ?o .
}
